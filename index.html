    <script>

        // Update clock every second

        function updateClock() {

            const now = new Date();

            document.getElementById('current-date').innerText = now.toLocaleDateString('en-US', { weekday: 'long', month: 'long', day: 'numeric' });

            document.getElementById('current-time').innerText = now.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' });

        }

        setInterval(updateClock, 1000);

        updateClock();


        // Tab Management

        function showTab(tab) {

            document.getElementById('home-tab').classList.toggle('hidden', tab !== 'home');

            document.getElementById('records-tab').classList.toggle('hidden', tab !== 'records');

            if (tab === 'records') loadRecords();

        }


        // Speech Recognition Setup

        const recordBtn = document.getElementById('record-btn');

        const btnText = document.getElementById('btn-text');

        const statusMsg = document.getElementById('status-msg');

        const liveTranscript = document.getElementById('live-transcript');

        const transcriptText = document.getElementById('transcript-text');


        let isRecording = false;

        let recognition;

        let finalTranscript = "";


        if ('webkitSpeechRecognition' in window) {

            recognition = new webkitSpeechRecognition();

            recognition.continuous = true;

            recognition.interimResults = true;


            recognition.onresult = (event) => {

                let interim = "";

                for (let i = event.resultIndex; i < event.results.length; ++i) {

                    if (event.results[i].isFinal) {

                        finalTranscript += event.results[i][0].transcript;

                    } else {

                        interim += event.results[i][0].transcript;

                    }

                }

                transcriptText.innerText = finalTranscript + interim;

            };


            recognition.onerror = (event) => {

                console.error(event.error);

                stopRecordingManually();

            };

        }


        function toggleRecording() {

            if (!isRecording) {

                startRecording();

            } else {

                stopRecording();

            }

        }


        function startRecording() {

            finalTranscript = "";

            transcriptText.innerText = "";

            isRecording = true;

            recordBtn.classList.add('recording');

            btnText.innerText = "STOP";

            statusMsg.innerText = "Listening to doctors' notes...";

            liveTranscript.classList.remove('hidden');

            recognition.start();

        }


        function stopRecording() {

            recognition.stop();

            saveRecord();

            stopRecordingUI();

        }


        function stopRecordingUI() {

            isRecording = false;

            recordBtn.classList.remove('recording');

            btnText.innerText = "START";

            statusMsg.innerText = "Note saved successfully.";

            setTimeout(() => { liveTranscript.classList.add('hidden'); }, 3000);

        }


        function saveRecord() {

            const docName = document.getElementById('doc-name').value || "Unknown Staff";

            const newRecord = {

                id: Date.now(),

                date: new Date().toLocaleDateString(),

                time: new Date().toLocaleTimeString(),

                doctor: docName,

                text: finalTranscript

            };


            const records = JSON.parse(localStorage.getItem('icu_notes') || "[]");

            records.unshift(newRecord);

            localStorage.setItem('icu_notes', JSON.stringify(records));

            document.getElementById('doc-name').value = ""; // Reset name

        }


        function loadRecords() {

            const container = document.getElementById('records-list');

            const records = JSON.parse(localStorage.getItem('icu_notes') || "[]");

            

            if (records.length === 0) {

                container.innerHTML = `<p class="text-gray-400 text-center py-12">No records yet.</p>`;

                return;

            }


            container.innerHTML = records.map(r => `

                <div class="bg-white p-5 rounded-2xl shadow-sm border border-gray-100">

                    <div class="flex justify-between items-start mb-2">

                        <div>

                            <h3 class="font-bold text-gray-800">${r.doctor}</h3>

                            <p class="text-xs text-gray-400">${r.date} at ${r.time}</p>

                        </div>

                    </div>

                    <p class="text-gray-600 text-sm leading-relaxed">${r.text || "No speech detected."}</p>

                </div>

            `).join('');

        }
